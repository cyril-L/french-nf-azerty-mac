```python
def get_spacing_clone(dead_key, prefer_second=True, combine_with='\u00A0'):
    clones = spacing_clones[dead_key]
    if prefer_second and clones[1] != None:
        return clones[1]
    elif clones[0] != None:
        return clones[0]
    else:
        return combine_with + dead_key[1]
```
